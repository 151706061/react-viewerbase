# Icon

## Basic usage

<Playground>
  <div
    style={{
      display: 'flex',
      flexWrap: 'wrap',
      alignItems: 'center',
      justifyContent: 'center',
      color: 'white',
    }}
  >
    {Object.keys(ICONS)
      .sort()
      .map((name, i) => {
        return (
          <React.Fragment key={i}>
            <div
              style={{
                display: 'flex',
                flexDirection: 'column',
                flexBasis: '25%',
                minWidth: '100px',
                margin: '16px',
                backgroundColor: 'rgb(11, 95, 255)',
                borderRadius: '8px',
                padding: '16px',
                alignItems: 'center',
              }}
            >
              <Icon name={name} />
              <div style={{ fontSize: '12px', wordWrap: 'break-word' }}>
                {name}
              </div>
            </div>
          </React.Fragment>
        );
      })}
  </div>
</Playground>
