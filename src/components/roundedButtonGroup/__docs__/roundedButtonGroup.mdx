# Rounded Button Group

A basic styled toggle switch

## Basic usage

<Playground>
  <State
    initial={{
      value: null,
      options: [
        {
          value: 'studies',
          icon: {
            name: 'studies',
            width: '15px',
            height: '13px',
          },
          bottomLabel: 'Series',
        },
        {
          value: 'measurements',
          icon: {
            name: 'measurement-lesions',
            width: '18px',
            height: '10px',
          },
          bottomLabel: 'Measurements',
        },
      ],
    }}
  >
    {({ state, setState }) => (
      <React.Fragment>
        <div>
          <pre>value: {JSON.stringify(state.value, null, 2)}</pre>
        </div>
        <RoundedButtonGroup
          {...state}
          onValueChanged={value => setState({ value })}
        />
      </React.Fragment>
    )}
  </State>
</Playground>
